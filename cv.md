Is the number prime?

```
function isPrime(num) {
  if (num % 1 !== 0) {
    return false;
  }
  if (num <= 1) {
    return false;
  }
  if (num <= 3) {
    return true;
  }
  if (num % 2 === 0) {
    return false;
  }
  const divLimit = Math.sqrt(num);
  for (let div = 3; div <= divLimit; div += 2) {
    if (num % div === 0) {
      return false;
    }
  }
  return true;
}
```
